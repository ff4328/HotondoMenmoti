SceneMain::SceneMain() :
    m_firstFrame(false),
    m_bossDead(false),
    m_playerDead(false),
    m_Pause(false),
    m_pPlayer(nullptr),
    m_pEnemy(nullptr),
    m_collision(nullptr),
    m_Items(nullptr)
{
    m_pPlayer = new PlayerMove();
    m_pEnemy = new EnemyYama();
    m_collision = std::make_unique<Collision>();
    m_Items = std::make_unique<Items>(m_pPlayer);
}
void SceneMain::Init()
{
    m_pPlayer->Init();
    m_pEnemy->Init();
    m_Items->Init();
}
void SceneMain::End()
{
    m_pPlayer->End();
    delete m_pPlayer;
    m_pPlayer = nullptr;
    m_pEnemy->End();
    delete m_pEnemy;
    m_pEnemy = nullptr;
    m_Items->End();
}

SceneBase* SceneMain::Update()
{
    m_pPlayer->Update();
    m_pEnemy->Update();
    m_Items->Update();
    return this;
}

void SceneMain::Draw()
{
    m_pPlayer->Draw();
    m_pEnemy->Draw();
    m_Items->Draw();
    DrawFade();
}
